406  }

408 
409 #ifdef USE_VCOM

411  printf("A data to transmit: [");

423

432  cCsState = S2LPQiGetCs();
433 
434  if(cCsState)
435  {

438  printf("RSSI: %.2f\n\n\r", (float)S2LPRadioGetRssidBm());
439 #endif
440  while(!xTxDoneFlag);
441  }
442  }
443  xTxDoneFlag = RESET;
444 
445  /* pause between RX and TX */
446  SdkDelayMs(500);
