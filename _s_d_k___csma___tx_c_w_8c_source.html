201 
202  /* S2LP Radio set power */
203  S2LPRadioSetMaxPALevel(S_DISABLE);
204  if(!S2LPManagementGetRangeExtender())
205  {
206  /* if we haven't an external PA, use the library function */
207  S2LPRadioSetPALeveldBm(7,POWER_DBM);
208  }
209  else
210  {
211  /* in case we are using the PA board, the S2LPRadioSetPALeveldBm will be not functioning because
212  the output power is affected by the amplification of this external component.
213  Set the raw register. */
214  uint8_t paLevelValue=0x25; /* for example, this value will give 23dBm about */
215  S2LPSpiWriteRegisters(PA_POWER8_ADDR, 1, &paLevelValue);
216  }
217  S2LPRadioSetPALevelMaxIndex(7);
218 
219  /* Spirit direct Tx mode - we have to configure a direct mode
220  that doesn't use the FIFO in order to not have an autoabort
221  during the transmission */
222  S2LPPacketHandlerSetTxMode(PN9_TX_MODE);
223 
224  /* TX command */

226 
227  /* infinite loop */
