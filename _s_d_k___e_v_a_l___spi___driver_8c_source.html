295  uint8_t rx_buff[255];

297 
298  SPI_ENTER_CRITICAL();
299 
300  /* Puts the SPI chip select low to start the transaction */
301  SdkEvalSPICSLow();
302 
303  HAL_SPI_TransmitReceive(&SpiHandle, tx_buff, rx_buff, 2, 1000);
304  HAL_SPI_TransmitReceive(&SpiHandle, pcBuffer, &rx_buff[2], cNbBytes, 1000);
305 
306  /* Puts the SPI chip select high to end the transaction */
307  SdkEvalSPICSHigh();
308 
309  SPI_EXIT_CRITICAL();
