181 uint16_t crcCalc(uint16_t crcReg, uint8_t crcData)
182 {
183  uint8_t i;
184 
185  for (i = 0; i < 8; i++)
186  {
187  // If upper most bit is 1
188  if (((crcReg & 0x8000) >> 8) ^ (crcData & 0x80))
189  crcReg = (crcReg << 1) ^ CRC_POLYNOM;
190  else
191  crcReg = (crcReg << 1);
192 
193  crcData <<= 1;
194  }
195  return crcReg;
196 }

212 void fitTxBuffer()
213 {
214  uint16_t nCrc, j=0, maxIndex, nFittedData;
215 
216  vectcTxBuff[j++] = cLField; // L field
217 
218  for(uint8_t i=0 ; i<9 ; i++)
219  {
220  vectcTxBuff[i+j] = i+1;
221  }
222 
223  nCrc = computeCrc(&vectcTxBuff[j-1],10);
224  j+=9;
225 
226  vectcTxBuff[j++] = (uint8_t)(nCrc>>8);
227  vectcTxBuff[j++] = (uint8_t) nCrc;
228  nFittedData=9;
229 
230  while(nFittedData<cLField)
231  {
232  if((cLField-nFittedData)/16 != 0)
233  maxIndex = 16;
234  else
235  maxIndex = (cLField-nFittedData)%16;
236 
237  for(uint8_t i=0 ; i<maxIndex ; i++)
238  {
239  vectcTxBuff[j++] = i+1;
240  nFittedData++;
241  }
242 
243  nCrc = computeCrc(&vectcTxBuff[j-maxIndex],maxIndex);
244  vectcTxBuff[j++] = (uint8_t)(nCrc>>8);
245  vectcTxBuff[j++] = (uint8_t) nCrc;
246  }

257 void HAL_GPIO_EXTI_Callback(uint16_t GPIO_Pin)
258 {
259  if(GPIO_Pin==M2S_GPIO_PIN_IRQ)
260  {
261 
262  /* Get the IRQ status */
263  S2LPGpioIrqGetStatus(&xIrqStatus);
264 
265 
266  /* Check the S2LP TX_DATA_SENT IRQ flag */
267  if(xIrqStatus.IRQ_TX_DATA_SENT)
268  {
269  /* set the xTxDoneFlag to manage the event in the main() */
270  xTxDoneFlag = SET;
271 
272  SdkEvalLedToggle(LED1);
273  }
274  else
275  /* Check the S2LP TX_FIFO_ALMOST_EMPTY IRQ flag */
276  if(xIrqStatus.IRQ_TX_FIFO_ALMOST_EMPTY)
277  {
278  /* read the number of elements in the Tx FIFO */
279  uint8_t cNElemTxFifo = S2LPFifoReadNumberBytesTxFifo();
280 
281  /* check if the sum of the residual payload to be transmitted and the actual bytes in FIFO are higher than 96 */
282  if(nResidualPcktLength+cNElemTxFifo>96)
283  {
284  /* .. if yes another almost full IRQ has to be managed */
285  /* ..so fill the Tx FIFO */
286  S2LPSpiWriteFifo(96-cNElemTxFifo-1, &vectcTxBuff[nTxIndex]);
287 
288  /* update the number of bytes to be transmitted */
289  nResidualPcktLength -= 96-cNElemTxFifo-1;
290 
291  /* update the number of bytes transmitted until now */
292  nTxIndex += 96-cNElemTxFifo-1;
293 
294  }
295  else
296  {
297  /* .. if not all the nResidualPcktLength bytes remaining can be written to the Tx FIFO */
298  /* ..so disable the TX_FIFO_ALMOST_EMPTY IRQ */
299  S2LPGpioIrqConfig(TX_FIFO_ALMOST_EMPTY , S_DISABLE);
300 
301  /* fill the Tx fifo */
302  S2LPSpiWriteFifo(nResidualPcktLength, &vectcTxBuff[nTxIndex]);
303 
304  /* unarm the AE threshold mechanism */
305  S2LPFifoSetAlmostEmptyThresholdTx(96);
306 
307  /* update the number of bytes to be transmitted */
308  nResidualPcktLength = 0;
309 
310  /* update the number of transmitted bytes */
311  nTxIndex += nResidualPcktLength;
312  }
313 
314  /* re-read the number of elements in the Tx FIFO */
315  cNElemTxFifo = S2LPFifoReadNumberBytesTxFifo();
316 
317  }
318  }
