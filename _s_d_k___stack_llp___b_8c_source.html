162 void HAL_GPIO_EXTI_Callback(uint16_t GPIO_Pin)
163 {
164  if(GPIO_Pin==M2S_GPIO_PIN_IRQ)
165  {
166  /* Get the IRQ status */
167  S2LPGpioIrqGetStatus(&xIrqStatus);
168 
169  /* Check the SPIRIT RX_DATA_DISC IRQ flag */
170  if(xIrqStatus.IRQ_RX_DATA_DISC)
171  {
172  S2LPCmdStrobeRx();
173  SdkEvalLedToggle(LED1);
174  }
175 
176  /* Check the SPIRIT TX_DATA_SENT IRQ flag for the acknoledgement */
177  if(xIrqStatus.IRQ_TX_DATA_SENT)
178  {
179 // SdkEvalLedToggle(LED1);
180 
181 #ifdef PIGGYBACKING
182  S2LPSpiWriteFifo(20, vectcTxBuff);
183  printf("Loaded piggybacking data: [");
184 
185  for(uint8_t i=0 ; i<20 ; i++)
186  printf("%d ", vectcTxBuff[i]);
187  printf("]\n\r");
188 #endif
189 
190  S2LPCmdStrobeRx();
191  }
192 
193  /* Check the SPIRIT RX_DATA_READY IRQ flag */
194  if(xIrqStatus.IRQ_RX_DATA_READY)
195  {
196  /* Get the RX FIFO size */
197  cRxData=S2LPFifoReadNumberBytesRxFifo();
198 
199  /* Read the RX FIFO */
200  S2LPSpiReadFifo(cRxData, vectcRxBuff);
201 
202  /* Flush the RX FIFO */
203  S2LPCmdStrobeFlushRxFifo();
204 
205  /* A simple way to control if the received data sequence is correct (in this case LED2 will toggle) */
206  {
207  int correct=0;
208 
209  for(uint8_t i=0 ; i<cRxData ; i++)
210  if(vectcRxBuff[i] != i+1)
211  correct=1;
212 
213  if(correct)
214  SdkEvalLedToggle(LED2);
215  }
216 #ifdef USE_VCOM
217  /* print the received data */
218  printf("B data received: [");
219 
220  for(uint8_t i=0 ; i<cRxData ; i++)
221  printf("%d ", vectcRxBuff[i]);
222  printf("]\r\n");
223 
224 #endif
225  if(S2LPPktStackGetTXAckRequest()) {
226  S2LPCmdStrobeRx();
227  }
228  }
229 
230  }
