185 uint16_t crcCalc(uint16_t crcReg, uint8_t crcData)
186 {
187  uint8_t i;
188 
189  for (i = 0; i < 8; i++)
190  {
191  // If upper most bit is 1
192  if (((crcReg & 0x8000) >> 8) ^ (crcData & 0x80))
193  crcReg = (crcReg << 1) ^ CRC_POLYNOM;
194  else
195  crcReg = (crcReg << 1);
196 
197  crcData <<= 1;
198  }
199 
200  return crcReg;
201 }

212 SBool CRCCheck(uint8_t *pStart, uint8_t *pStop)
213 {
214  uint16_t seed= 0x0000;
215 
216  while (pStart != pStop)
217  {
218  seed = crcCalc(seed, *pStart);
219  pStart++;
220  }
221  seed =~seed;
222  if((pStop[0] == (uint8_t)(seed>>8))&&(pStop[1] == (uint8_t)(seed)))
223  {
224  return S_TRUE;
225  }
226  return S_FALSE;
227 }

233 static void packetReceptionSM(void)
234 {
235  static uint8_t rx_state=SM_RX_NEW_PACKET;
236  static uint16_t nResidualPcktLength,nRxIndex;
237  static uint8_t cRxDataLen;
238  uint8_t cThreholdRxFifoAF;
239 
240  switch(rx_state)
241  {
242 
243  case SM_RX_NEW_PACKET:
244  {
245 
246  S2LPSpiReadFifo(12, &vectcRxBuff[0]);
247  nRxIndex=12;
248  uint8_t cLField=vectcRxBuff[0];
249 
250  /* L should be at least 10 (C-field+M-field+A-field+CI-field), otherwise give an error */
251  if(cLField<10)
252  {
253  S2LPGpioIrqClearStatus();
254  S2LPCmdStrobeSabort();
255  S2LPCmdStrobeFlushRxFifo();
256  S2LPCmdStrobeRx();
257  /* The SM should not continue! */
258  return;
259  }
260 
261 
262  nPcktLength = 1+cLField+2+2*(CEILING(((float)cLField-9)/16));
263 
264 
265  nResidualPcktLength=nPcktLength-nRxIndex;
266 
267  if(nResidualPcktLength>94)
268  {
269  cThreholdRxFifoAF = 48;
270  rx_state = SM_RX_INTERM;
271  }
272  else
273  {
274  cThreholdRxFifoAF = nResidualPcktLength;
275  rx_state = SM_RX_LAST;
276  }
277  S2LPFifoSetAlmostFullThresholdRx(96-cThreholdRxFifoAF);
278 
279 
280  }
281  break;
282  case SM_RX_LAST:
283  {
284  /* stop the reception*/
285  S2LPCmdStrobeSabort();
286 
287  /* unarm the almost full threshold */
288  S2LPGpioIrqConfig(RX_FIFO_ALMOST_FULL, S_DISABLE);
289 
290  /* Read the RX FIFO */
291  S2LPSpiReadFifo(nResidualPcktLength, &vectcRxBuff[nRxIndex]);
292  /* update the number of received bytes */
293  nResidualPcktLength=nPcktLength-nRxIndex-nResidualPcktLength;
294 
295  S2LPCmdStrobeFlushRxFifo();
296 
297  S2LPFifoSetAlmostFullThresholdRx(96-12);
298 
299 
300  rx_state=SM_RX_NEW_PACKET;
301  xRxDone=S_SET;
302 
303  }
304  break;
305  case SM_RX_INTERM:
306  {
307 
308  /* we read always the same quantity of bytes */
309  cRxDataLen = 47;
310 
311  S2LPSpiReadFifo(cRxDataLen, &vectcRxBuff[nRxIndex]);
312 
313  nRxIndex+=cRxDataLen;
314  nResidualPcktLength=nPcktLength-nRxIndex;
315 
316 
317  if(nResidualPcktLength<90)
318  {
319  S2LPFifoSetAlmostFullThresholdRx(96-nResidualPcktLength);
320  rx_state = SM_RX_LAST;
321  }
322 
323  /* workaround... because sometimes the IRQ line is found low because AF IRQ */
324  S2LPIrqs irqStatus; S2LPGpioIrqGetStatus(&irqStatus);
325 
326  }
327 
328  break;
329 
330  }
